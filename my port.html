// CONTACT FORM HANDLER - REPLACE THE EXISTING CONTACT FORM CODE const contactForm = document.getElementById('contactForm'); const statusDiv = document.getElementById('form-status'); contactForm.addEventListener('submit', async function(e) { e.preventDefault(); // Show loading state statusDiv.innerHTML = ' Sending message...'; statusDiv.style.display = 'block'; // Collect form data with proper field names const formData = new FormData(contactForm); // Add your Web3Forms access key here formData.append('access_key', 'YOUR_WEB3FORMS_ACCESS_KEY'); // Add redirect URL (optional) - set to false to stay on page formData.append('redirect', 'false'); // Add from name for better email presentation formData.append('from_name', 'ShopifyX Portfolio'); try { const response = await fetch('https://api.web3forms.com/submit', { method: 'POST', body: formData }); const result = await response.json(); if (result.success) { statusDiv.innerHTML = ` ✅ Message sent successfully! I'll get back to you within 2-4 hours. `; contactForm.reset(); // Auto-hide success message after 8 seconds setTimeout(() => { statusDiv.style.opacity = '0'; setTimeout(() => { statusDiv.style.display = 'none'; statusDiv.style.opacity = '1'; }, 500); }, 8000); } else { statusDiv.innerHTML = ` ❌ ${result.message || 'Error sending message. Please try again.'} `; } } catch (error) { console.error('Web3Forms error:', error); statusDiv.innerHTML = ` ❌ Network error. Please check your connection and try again. `; } });
